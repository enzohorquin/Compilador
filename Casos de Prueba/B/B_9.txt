x:uinteger.
result:double.
x=2.
SWITCH(x){
	CASE 1: 
		BEGIN 
			IF(x>0)THEN
				BEGIN
					result=1,0.OUT("1").
				END.
			ELSE
				BEGIN
					result=2,0.OUT("2").
				END.
			ENDIF
		END..
	CASE 2: 
	BEGIN 
		IF(x>1)THEN
			BEGIN
				result=2,3.OUT("3").
			END.
		ELSE
			BEGIN
				result=3,1.OUT("4").
			END.
		ENDIF
	END..
}.
